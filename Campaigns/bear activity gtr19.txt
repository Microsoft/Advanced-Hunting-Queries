ProcessCreationEvents 
| where ((FolderPath endswith "\\xcopy.exe" and ProcessCommandLine has " /S /E /C /Q /H \\") or (FolderPath endswith "\\adexplorer.exe" and ProcessCommandLine has " -snapshot \"\" c:\\users\\"))
| where EventTime > ago(7d)
